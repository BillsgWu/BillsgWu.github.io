所以我们应该先定义一个函数

template<typename Type>//泛型的声明,没这句会报错
Type abs(Type num)//定义绝对值函数
{
    if(num>0)//如果是正数……
    {
        return num;//原样退货
    }
    else//如果是非正数(负数和零)……
    {
        return 0-num;//改改后退货
    }
}
